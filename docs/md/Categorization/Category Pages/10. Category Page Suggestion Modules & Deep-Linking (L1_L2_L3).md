**Rules**

- Only **published** categories render on the public website.

- **L2** are subcategories of **L1**; **L3** are subcategories of
  **L2**.

- L3 pages are reached by navigating **L1 → L2 → L3** (these are
  **pages, not tabs**).

## **Step 2 --- Embeddings pipeline for "Related Categories" (Hugging Face)**

**Explanation\**
Precompute sentence embeddings for each category (name + short
description + ancestors) using sentence-transformers/all-MiniLM-L6-v2
(384‑dim). Store vectors in Postgres via pgvector and query by cosine
similarity. This is a **local** component for Category Pages.

### **2.1 DB migration (pgvector + table)**

**Folder path\**
/apps/api/src/db/migrations/2025_10_02_pgvector_categories.sql

CREATE TABLE IF NOT EXISTS category_embedding (

category_uuid UUID PRIMARY KEY REFERENCES categories(uuid) ON DELETE
CASCADE,

embed vector(384) NOT NULL,

updated_at timestamptz NOT NULL DEFAULT now()

);

\-- Approximate NN index for cosine distance

CREATE INDEX IF NOT EXISTS idx_category_embed_ivf

ON category_embedding USING ivfflat (embed vector_cosine_ops) WITH
(lists = 100);

import { Client } from \'pg\'

import { SentenceTransformer } from \'sentence-transformers\' // if not
available, use the Python worker variant

const modelName = \'sentence-transformers/all-MiniLM-L6-v2\'

async function main() {

const pg = new Client({ connectionString: process.env.DATABASE_URL! })

await pg.connect()

// Categories needing (re)embedding --- updated since last embedding

const { rows: cats } = await pg.query(\`

SELECT c.uuid, c.name, COALESCE(c.short_description,\'\') AS
short_description,

l1.slug AS l1, l2.slug AS l2, l3.slug AS l3

FROM categories c

LEFT JOIN categories l3 ON l3.uuid = c.uuid

LEFT JOIN categories l2 ON l2.uuid = c.parent_uuid

LEFT JOIN categories l1 ON l1.uuid = l2.parent_uuid

WHERE c.updated_at \> COALESCE((SELECT updated_at FROM
category_embedding e WHERE e.category_uuid = c.uuid), \'1970-01-01\')

\`)

const model = await SentenceTransformer.load(modelName)

for (const c of cats) {

const parts = \[c.l1, c.l2, c.l3\].filter(Boolean).join(\' \> \')

const text = \`\${parts ? parts + \' --- \' :
\'\'}\${c.name}\${c.short_description ? \' --- \' + c.short_description
: \'\'}\`

const vec = await model.encode(\[text\]) as Float32Array // 384‑dim

await pg.query(

\`INSERT INTO category_embedding(category_uuid, embed)

VALUES (\$1, \$2)

ON CONFLICT (category_uuid) DO UPDATE SET embed = EXCLUDED.embed,
updated_at = now()\`,

\[c.uuid, vec\]

)

}

await pg.end()

}

main().catch(err =\> { console.error(err); process.exit(1) })

## **Step 3 --- Repo helpers (pgvector NN + heuristics)**

**Explanation\**
Add thin queries for: nearest neighbors (semantic), trending, popular,
and skills. Enforce **published‑only**.

**Folder path\**
/apps/api/src/categories/categories.repo.ts

export class CategoriesRepo {

constructor(private readonly db: { query\<T=any\>(sql:string,
params?:any\[\]): Promise\<{ rows:T\[\] }\> }) {}

async nodeFor(ctx: { l1?: string; l2?: string; l3?: string }) {

const slug = ctx.l3 \|\| ctx.l2 \|\| ctx.l1

if (!slug) return null

const { rows } = await this.db.query(\`SELECT uuid, slug, l1_slug FROM
categories WHERE slug = \$1 LIMIT 1\`, \[slug\])

return rows\[0\] \|\| null

}

async embeddingFor(uuid: string) {

const { rows } = await this.db.query(\`SELECT embed FROM
category_embedding WHERE category_uuid = \$1\`, \[uuid\])

return rows\[0\] \|\| null

}

async nearestNeighbors(opts: { uuid: string; l1Slug: string; limit:
number }) {

const { uuid, l1Slug, limit } = opts

const { rows } = await this.db.query(\`

WITH focus AS (SELECT embed FROM category_embedding WHERE category_uuid
= \$1)

SELECT c.uuid, c.name, c.slug, c.parent_uuid, p.parent_uuid AS
parent_of_parent,

c.published_in_marketplace,

ce.embed \<=\> (SELECT embed FROM focus) AS cos_dist

FROM categories c

JOIN category_embedding ce ON ce.category_uuid = c.uuid

LEFT JOIN categories p ON p.uuid = c.parent_uuid

WHERE c.published_in_marketplace = true

AND c.uuid \<\> \$1

AND c.l1_slug = \$2 \-- prefer same L1 subtree

ORDER BY cos_dist ASC

LIMIT \$3

\`, \[uuid, l1Slug, limit\])

return rows

}

// If subtree is sparse, second pass across all L1 (still
published‑only)

async nearestNeighborsGlobal(opts: { uuid: string; limit: number }) {

const { uuid, limit } = opts

const { rows } = await this.db.query(\`

WITH focus AS (SELECT embed FROM category_embedding WHERE category_uuid
= \$1)

SELECT c.uuid, c.name, c.slug, c.parent_uuid, p.parent_uuid AS
parent_of_parent,

c.published_in_marketplace,

ce.embed \<=\> (SELECT embed FROM focus) AS cos_dist

FROM categories c

JOIN category_embedding ce ON ce.category_uuid = c.uuid

LEFT JOIN categories p ON p.uuid = c.parent_uuid

WHERE c.published_in_marketplace = true

AND c.uuid \<\> \$1

ORDER BY cos_dist ASC

LIMIT \$2

\`, \[uuid, limit\])

return rows

}

async trendingForContext(\_: any) {

const { rows } = await this.db.query(\`

SELECT c.uuid, c.name, c.slug, c.parent_uuid, p.parent_uuid AS
parent_of_parent, c.published_in_marketplace

FROM categories c

LEFT JOIN categories p ON p.uuid = c.parent_uuid

WHERE c.published_in_marketplace = true

ORDER BY c.trending_score DESC NULLS LAST

LIMIT 24

\`)

return rows

}

async popularForContext(\_: any) {

const { rows } = await this.db.query(\`

SELECT c.uuid, c.name, c.slug, c.parent_uuid, p.parent_uuid AS
parent_of_parent, c.published_in_marketplace

FROM categories c

LEFT JOIN categories p ON p.uuid = c.parent_uuid

WHERE c.published_in_marketplace = true

ORDER BY c.impressions DESC NULLS LAST

LIMIT 24

\`)

return rows

}

async skillsForContext(\_: any) {

const { rows } = await this.db.query(\`

SELECT DISTINCT s.name, s.slug

FROM skills s

JOIN category_skills cs ON cs.skill_uuid = s.uuid

JOIN categories c ON c.uuid = cs.category_uuid

WHERE c.published_in_marketplace = true

ORDER BY s.popularity DESC NULLS LAST

LIMIT 30

\`)

return rows

}

async slugPathFor(uuid: string): Promise\<string\[\]\> {

const { rows } = await this.db.query(\`

WITH RECURSIVE chain AS (

SELECT uuid, slug, parent_uuid FROM categories WHERE uuid = \$1

UNION ALL

SELECT c.uuid, c.slug, c.parent_uuid FROM categories c

JOIN chain ON c.uuid = chain.parent_uuid

)

SELECT ARRAY_AGG(slug ORDER BY array_position(ARRAY(SELECT uuid FROM
chain), uuid) DESC) AS path

FROM chain

\`, \[uuid\])

return rows\[0\]?.path \|\| \[\]

}

}

## **Step 4 --- Suggestions Service (compose modules)**

**Explanation\**
Implements Related via embeddings NN (with a global fallback), and
Trending/Popular/Skills via repo helpers. Filters to published‑only and
builds deep links.

**Folder path\**
/apps/api/src/category-pages/suggestions.service.ts

import { Injectable } from \'@nestjs/common\'

import { CategoriesRepo } from \'../categories/categories.repo\'

import type { SuggestedCategory, SuggestedSkill } from \'./types\'

type Ctx = { l1?: string; l2?: string; l3?: string }

type RawCat = { uuid:string; name:string; slug:string;
parent_uuid?:string\|null; parent_of_parent?:boolean;
published_in_marketplace:boolean }

function levelOf(r: RawCat): 1\|2\|3 {

return !r.parent_uuid ? 1 : (r.parent_of_parent ? 3 : 2)

}

\@Injectable()

export class SuggestionsService {

constructor(private readonly repo: CategoriesRepo) {}

private async toSuggestedCategory(row: RawCat):
Promise\<SuggestedCategory\> {

const slugPath = await this.repo.slugPathFor(row.uuid)

return { uuid: row.uuid, level: levelOf(row), name: row.name, slugPath,
link: \`/c/\${slugPath.join(\'/\')}\` }

}

async related(ctx: Ctx): Promise\<SuggestedCategory\[\]\> {

const focus = await this.repo.nodeFor(ctx)

if (!focus) return \[\]

const embed = await this.repo.embeddingFor(focus.uuid)

if (!embed) return \[\]

// First pass: same L1 subtree

let rows = await this.repo.nearestNeighbors({ uuid: focus.uuid, l1Slug:
focus.l1_slug, limit: 24 })

// Fallback: allow global neighbors if sparse

if (rows.length \< 8) {

const extra = await this.repo.nearestNeighborsGlobal({ uuid: focus.uuid,
limit: 24 })

const seen = new Set(rows.map(r =\> r.uuid))

rows.push(\...extra.filter(e =\> !seen.has(e.uuid)))

}

return Promise.all(rows.map(r =\> this.toSuggestedCategory(r)))

}

async trending(ctx: Ctx): Promise\<SuggestedCategory\[\]\> { return
Promise.all((await this.repo.trendingForContext(ctx)).map(r =\>
this.toSuggestedCategory(r))) }

async popular(ctx: Ctx): Promise\<SuggestedCategory\[\]\> { return
Promise.all((await this.repo.popularForContext(ctx)).map(r =\>
this.toSuggestedCategory(r))) }

async relatedSkills(\_: Ctx): Promise\<SuggestedSkill\[\]\> { const rows
= await this.repo.skillsForContext({}); return rows.map((s: any) =\> ({
name: s.name, slug: s.slug, link:
\`/skills/\${encodeURIComponent(s.slug)}\` })) }

}

## **Step 8 --- QA**

1.  **Semantic Related works\**
    • On an L2 page, "Related Categories" reads like true semantic
    neighbors (same L1 first; cross‑L1 only if local is sparse).

2.  **Published‑only\**
    • Unpublish a category; it never appears in any module.

3.  **No broken links\**
    • All suggestions link to valid /c/:l1, /c/:l1/:l2, or
    /c/:l1/:l2/:l3 pages; page‑health prevents empty pages.

4.  **Performance\**
    • Embeddings precomputed; pgvector NN returns results within target
    latency (tune lists, memory).
