**Default meta templates, override fields in Strapi, JSON-LD
breadcrumbs, canonical tags, OpenGraph/Twitter**

> (subcategory of L2).\

## **Step 2 --- NestJS SEO builder (defaults + overrides + JSON-LD)**

**Explanation\**
This function **does not replace** your categorization SEO logic; it
**wraps/reuses** it for **category pages** so canonical/JSON-LD and
per-page variants (All/Projects/Services/Teams/Talent) are correct.

**Folder path\**
/apps/api/src/seo/buildSeo.ts *(new, additive)*

**// /apps/api/src/seo/buildSeo.ts

type Crumb = { name: string; slug: string; uuid: string }

export type PageKind = \'show-all\' \| \'projects\' \| \'services\' \|
\'teams\' \| \'talent\'

export type SeoPayload = {

title: string

description: string

canonical: string

noindex: boolean

og: { title: string; description: string; url: string; image?: string }

twitter: { card: \'summary_large_image\' \| \'summary\'; title: string;
description: string; image?: string }

breadcrumbJsonLd: any

}

export async function buildCategorySeo({

siteName,

origin,

slugPath,

breadcrumbs,

pageKind,

overrides // values fetched from Strapi Category.seo (or null)

}: {

siteName: string

origin: string

slugPath: string\[\]

breadcrumbs: Crumb\[\]

pageKind: PageKind

overrides?: {

metaTitle?: string \| null

metaDescription?: string \| null

noindex?: boolean \| null

ogImageUrl?: string \| null

twitterImageUrl?: string \| null

} \| null

}): Promise\<SeoPayload\> {

// Defaults derived from category name + page kind

const lastName = (breadcrumbs\[breadcrumbs.length - 1\]?.name \|\|
\'Categories\').trim()

const kindSuffix = pageKind === \'show-all\' ? \'\' : \` --
\${cap(pageKind)}\`

const defaultTitle = \`\${lastName}\${kindSuffix} \| \${siteName}\`

const defaultDesc = \`Explore \${lastName} across projects, services,
teams and talent.\`

// Canonical includes page kind when not \"show-all\"

const canonical = \`\${origin}/c/\${slugPath.join(\'/\')}\${pageKind ===
\'show-all\' ? \'\' : \`/\${pageKind}\`}\`

// Apply overrides (if any)

const title = overrides?.metaTitle?.trim() \|\| defaultTitle

const description = overrides?.metaDescription?.trim() \|\| defaultDesc

const noindex = !!overrides?.noindex

// OG/Twitter images (Twitter falls back to OG if not set)

const ogImg = overrides?.ogImageUrl \|\| undefined

const twImg = overrides?.twitterImageUrl \|\| ogImg \|\| undefined

const breadcrumbJsonLd = {

\'@context\': \'https://schema.org\',

\'@type\': \'BreadcrumbList\',

\'itemListElement\': breadcrumbs.map((b, i) =\> ({

\'@type\': \'ListItem\',

\'position\': i + 1,

\'name\': b.name,

\'item\': \`\${origin}/c/\${breadcrumbs.slice(0, i + 1).map(x =\>
x.slug).join(\'/\')}\`

}))

}

return {

title,

description,

canonical,

noindex,

og: { title, description, url: canonical, image: ogImg },

twitter: {

card: twImg ? \'summary_large_image\' : \'summary\',

title,

description,

image: twImg

},

breadcrumbJsonLd

}

}

function cap(s: string) { return s.charAt(0).toUpperCase() + s.slice(1)
}

## **Step 4 --- Wire SEO into Category Page Assembler (no rebuild)**

**Explanation\**
Use the builder from Step 2 inside your **existing** Category Page
service. Don't change your assembler/evaluators---just set payload.seo.

**Folder path\**
/apps/api/src/category-pages/category-pages.service.ts *(add the
highlighted lines)*

**import { buildCategorySeo, PageKind } from \'../seo/buildSeo\'

export class CategoryPagesService {

// \...

async buildPagePayload(slugPath: string\[\], pageKind: PageKind):
Promise\<any \| null\> {

const resolved = await this.categories.resolvePath(slugPath)

if (!resolved \|\| !resolved.published) return null

const overrides = await
this.categories.fetchSeoOverrides(resolved.uuids\[\`l\${resolved.level}\`\])

const seo = await buildCategorySeo({

siteName: process.env.SITE_NAME \|\| \'Marketplace\',

origin: process.env.PUBLIC_WEB_ORIGIN!,

slugPath,

breadcrumbs: resolved.breadcrumbs,

pageKind,

overrides

})

const blocks = await this.assembler.composeBlocks(resolved.uuids,
resolved.level, await this.taxGroups.forPath(resolved.uuids,
resolved.level))

const subcategories = await
this.categories.childrenOf(resolved.uuids\[\`l\${resolved.level}\`\])

return {

level: resolved.level,

slugPath,

uuids: resolved.uuids,

breadcrumbs: resolved.breadcrumbs,

hero: resolved.hero \|\| null, // reuse existing

subcategories,

blocks,

seo // 👈 set the SEO object here

}

}

}

## **Step 5 --- Next.js: render SEO head + breadcrumbs JSON-LD (tiny component)**

**Explanation\**
Drop this once in your page; it reads payload.seo from the API. If you
already have a head component, just map these fields and **don't
duplicate**.

**Folder path\**
/apps/web/components/seo/SeoHead.tsx *(new or adapt your current head)*

**import Head from \'next/head\'

export function SeoHead({ seo }: {

seo: {

title: string

description: string

canonical: string

noindex?: boolean

og: { title: string; description: string; url: string; image?: string }

twitter: { card: \'summary_large_image\'\|\'summary\'; title: string;
description: string; image?: string }

breadcrumbJsonLd: any

}

}) {

return (

\<Head\>

\<title\>{seo.title}\</title\>

\<meta name=\"description\" content={seo.description} /\>

\<link rel=\"canonical\" href={seo.canonical} /\>

{seo.noindex && \<meta name=\"robots\" content=\"noindex,nofollow\" /\>}

\<meta property=\"og:title\" content={seo.og.title} /\>

\<meta property=\"og:description\" content={seo.og.description} /\>

\<meta property=\"og:url\" content={seo.og.url} /\>

{seo.og.image && \<meta property=\"og:image\" content={seo.og.image}
/\>}

\<meta name=\"twitter:card\" content={seo.twitter.card} /\>

\<meta name=\"twitter:title\" content={seo.twitter.title} /\>

\<meta name=\"twitter:description\" content={seo.twitter.description}
/\>

{seo.twitter.image && \<meta name=\"twitter:image\"
content={seo.twitter.image} /\>}

\<script type=\"application/ld+json\"

dangerouslySetInnerHTML={{ \_\_html:
JSON.stringify(seo.breadcrumbJsonLd) }} /\>

\</Head\>

)

}

## **Step 7 --- Webhooks & cache invalidation (reuse your pipeline)**

**Explanation\**
When editors change SEO in Strapi, your **existing** webhook/ISR/edge
purge should invalidate category pages. If you already have this for
categories, **no changes needed**; these pages will ride the same
mechanism.

## **Step 8 --- QA checklist**

- **Defaults**: With no overrides, API returns sensible
  title/description + canonical including page kind (except "All").

- **Overrides**: Add meta title/description + images in Strapi → API
  reflects; \<SeoHead\> renders OG/Twitter and robots noindex when
  toggled.

- **Breadcrumb JSON-LD**: Validates in Rich Results tester; each crumb
  links to /c/:path step.

- **Canonical**: /c/:path and /c/:path/:kind always show the expected
  \<link rel=\"canonical\"\>.

- **No rebuild**: Verify you didn't duplicate existing categorization
  SEO code---only mapped/wired what was missing for category pages.
