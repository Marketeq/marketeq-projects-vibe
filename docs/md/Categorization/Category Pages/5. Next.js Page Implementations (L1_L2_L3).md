# **Doc 5 --- Next.js Page Implementations (L1/L2/L3)**

**Goal:** Build /c/\[\...slug\] to render L1 (top-level), L2
(subcategory of L1), and L3 (subcategory of L2) pages.

- First load uses **SSR** for SEO.

- Switching **views** (All, Projects, Services, Teams, Talent, or "Most
  Popular/Recent"---whatever the design dictates) updates **only the
  blocks area** with client-side fetch (no full page reload).

- Clicking a **subcategory** (L1→L2 or L2→L3) **navigates to a new
  page** (that is a new category level).

- Includes **skeleton loaders**, **RTL** support hook, and a clean
  separation of concerns.

## **Step 1 --- Page route with SSR + client re-hydration**

**Explanation\**
We serve the first request via SSR for SEO (meta, canonical,
breadcrumbs). After hydration, we keep the shell (header, breadcrumbs,
subcategory list) and only swap the **blocks area** when a user changes
a view. The API used is /api/category-pages/:slug\[/:view?\] from Doc 4.

**Folder path\**
/apps/web/pages/c/\[\...slug\].tsx

import Head from \'next/head\'

import { GetServerSideProps } from \'next\'

import { useRouter } from \'next/router\'

import { useEffect, useMemo, useState } from \'react\'

import { useRtl } from \'@/components/i18n/useRtl\'

import { Blocks } from \'@/components/category/Blocks\'

\'@/components/category/SubcategoryList\'

import { ShellSkeleton } from \'@/components/category/ShellSkeleton\'

import { BlocksSkeleton } from \'@/components/category/BlocksSkeleton\'

export type PageView = \'show-all\' \| \'projects\' \| \'services\' \|
\'teams\' \| \'talent\'

export type PagePayload = {

level: 1\|2\|3

slugPath: string\[\]

uuids: { l1: string; l2?: string; l3?: string }

breadcrumbs: Array\<{ name: string; slug: string; uuid: string }\>

seo: { title: string; description: string; canonical: string }

hero?: { title: string; image?: string; link?: string } \| null

subcategories: Array\<{ name: string; slug: string; uuid: string }\>

blocks: Array\<{

key: string

type: \'grid\'\|\'carousel\'

entity:
\'projects\'\|\'services\'\|\'teams\'\|\'talent\'\|\'guides\'\|\'courses\'

items: any\[\]

pagination?: { nextCursor?: string }

}\>

}

type Props = { initialPayload: PagePayload, initialView: PageView }

export default function CategoryPage({ initialPayload, initialView }:
Props) {

useRtl() // enables dir=\"rtl\" if needed

const router = useRouter()

const \[payload, setPayload\] = useState\<PagePayload\>(initialPayload)

const \[view, setView\] = useState\<PageView\>(initialView)

const \[loadingBlocks, setLoadingBlocks\] = useState(false)

const slugPath = useMemo(() =\> (router.query.slug as string\[\]) \|\|
\[\], \[router.query.slug\])

const baseHref = \`/c/\${slugPath.join(\'/\')}\`

// Change view without full reload: shallow routing + dynamic fetch

const handleChangeView = async (nextView: PageView) =\> {

if (nextView === view) return

setView(nextView)

setLoadingBlocks(true)

const url =
\`\${process.env.NEXT_PUBLIC_API_URL}/category-pages/\${slugPath.join(\'/\')}\${nextView===\'show-all\'
? \'\' : \`/\${nextView}\`}\`

const res = await fetch(url, { headers: { \'x-edge-cache\': \'allow\' }
})

if (res.ok) {

const json = await res.json()

setPayload((prev) =\> ({ \...prev, blocks: json.blocks, seo: json.seo
})) // keep shell; swap blocks/seo if needed

}

setLoadingBlocks(false)

// Shallow update URL so users can share link, but do not reload the
page

const dest = \`\${baseHref}\${nextView===\'show-all\' ? \'\' :
\`/\${nextView}\`}\`

router.push(dest, dest, { shallow: true })

}

if (!payload) return \<ShellSkeleton /\>

return (

\<\>

\<Head\>

\<title\>{payload.seo.title}\</title\>

\<meta name=\"description\" content={payload.seo.description} /\>

\<link rel=\"canonical\" href={payload.seo.canonical} /\>

\</Head\>

{/\* Hero (design-dependent) \*/}

{payload.hero && (

\<section className=\"container mx-auto px-4 py-6\"\>

\<div className=\"rounded-2xl border p-6 flex items-center
justify-between\"\>

\<div\>

\<h1 className=\"text-2xl font-semibold\"\>{payload.hero.title}\</h1\>

\</div\>

{payload.hero.image && (

// eslint-disable-next-line \@next/next/no-img-element

\<img src={payload.hero.image} alt=\"\" className=\"w-32 h-20
object-cover rounded-lg\" /\>

)}

\</div\>

\</section\>

)}

{/\* Breadcrumbs \*/}

\<nav aria-label=\"Breadcrumbs\" className=\"container mx-auto px-4 py-2
text-sm\"\>

\<ol className=\"flex gap-2\"\>

{payload.breadcrumbs.map((b, i) =\> (

\<li key={b.uuid} className=\"flex items-center gap-2\"\>

\<a className=\"underline\" href={\`/c/\${payload.breadcrumbs.slice(0,
i + 1).map(x =\> x.slug).join(\'/\')}\`}\>

{b.name}

\</a\>

{i \< payload.breadcrumbs.length - 1 && \<span\>/\</span\>}

\</li\>

))}

\</ol\>

\</nav\>

{/\* Subcategories (clicking goes deeper: L1→L2 or L2→L3 = full page
load) \*/}

{payload.subcategories?.length \> 0 && (

\<SubcategoryList currentPath={slugPath} subs={payload.subcategories}
/\>

)}

{/\* Views (inside the page): All / Projects / Services / Teams /
Talent. NO full reload. \*/}

\<section className=\"container mx-auto px-4 mt-4\"\>

\<ViewTabs

active={view}

onChange={handleChangeView}

makeHref={(v) =\> \`\${baseHref}\${v===\'show-all\' ? \'\' :
\`/\${v}\`}\`}

/\>

\</section\>

{/\* Blocks area \*/}

\<section className=\"container mx-auto px-4 py-6\"\>

{loadingBlocks ? \<BlocksSkeleton /\> : \<Blocks blocks={payload.blocks}
/\>}

\</section\>

\</\>

)

}

export const getServerSideProps: GetServerSideProps = async (ctx) =\> {

const slug = (ctx.params?.slug as string\[\]) \|\| \[\]

// Support direct entity view paths:
/c/:l1/:l2?/:l3?/(projects\|services\|teams\|talent)

const known: PageView\[\] =
\[\'projects\',\'services\',\'teams\',\'talent\'\]

const last = slug\[slug.length-1\]

const isEntityView = known.includes(last as any)

const slugPath = isEntityView ? slug.slice(0, -1) : slug

const initialView: PageView = isEntityView ? (last as PageView) :
\'show-all\'

const url =
\`\${process.env.NEXT_PUBLIC_API_URL}/category-pages/\${slugPath.join(\'/\')}\${initialView===\'show-all\'
? \'\' : \`/\${initialView}\`}\`

const res = await fetch(url)

if (res.status === 404) return { notFound: true }

const initialPayload: PagePayload = await res.json()

// Force canonical path segment form

const expectedPath =
\`/c/\${slugPath.join(\'/\')}\${initialView===\'show-all\' ? \'\' :
\`/\${initialView}\`}\`

if (ctx.resolvedUrl !== expectedPath) {

return { redirect: { destination: expectedPath, permanent: true } }

}

return { props: { initialPayload, initialView } }

}

## **Step 2 --- Reusable hooks (RTL + window-safe utilities)**

**Explanation\**

dir=\"rtl\" on \<html\> when needed. You can swap this for your i18n
solution later.

**Folder path\**

// /apps/web/components/i18n/useRtl.ts

import { useEffect } from \'react\'

export function useRtl(isRtl = false) {

useEffect(() =\> {

const html = document.documentElement

const prev = html.getAttribute(\'dir\') \|\| \'ltr\'

if (isRtl) html.setAttribute(\'dir\', \'rtl\')

return () =\> html.setAttribute(\'dir\', prev)

}, \[isRtl\])

}

## **Step 5 --- Blocks renderer (grids + carousels)**

**Explanation\**
Renders the dynamic blocks returned by the API. Sections that have

payload (we capped items to 10 "pages" worth).

function titleFor(entity: string) {

switch (entity) {

case \'projects\': return \'Projects\'

case \'services\': return \'Services\'

case \'teams\': return \'Teams\'

case \'talent\': return \'Talent\'

case \'guides\': return \'Guides\'

case \'courses\': return \'Courses\'

default: return \'Content\'

}

\<h3 className=\"font-medium\"\>{item.title \|\| item.name}\</h3\>

{/\* Add entity-specific fields here per your Level page designs \*/}

import { useRef, useState } from \'react\'

export function Carousel\<T\>({

items,

renderItem,

perView = 3

}: {

items: T\[\]

renderItem: (it: T) =\> JSX.Element

perView?: number

}) {

const trackRef = useRef\<HTMLDivElement\>(null)

const \[page, setPage\] = useState(0)

const pageCount = Math.ceil(items.length / perView)

const canPrev = page \> 0

const canNext = page \< (pageCount - 1)

const start = page \* perView

const visible = items.slice(start, start + perView)

return (

\<div className=\"relative\"\>

\<div className=\"flex gap-4 overflow-hidden\" ref={trackRef}\>

{visible.map((it, i) =\> (

\<div key={i} className=\"min-w-0 flex-1\"\>{renderItem(it)}\</div\>

))}

\</div\>

\<div className=\"mt-2 flex items-center justify-between\"\>

\<button disabled={!canPrev} onClick={() =\> setPage(p =\> Math.max(0,
p-1))}

className={\`px-2 py-1 border rounded \${!canPrev && \'opacity-40
cursor-not-allowed\'}\`}\>

←

\</button\>

\<div className=\"text-xs\"\>{page+1} / {pageCount}\</div\>

\<button disabled={!canNext} onClick={() =\> setPage(p =\>
Math.min(pageCount-1, p+1))}

className={\`px-2 py-1 border rounded \${!canNext && \'opacity-40
cursor-not-allowed\'}\`}\>

→

\</button\>

\</div\>

\</div\>

)

}

## **Step 7 --- Guardrails (what must NOT reload)**

**Explanation**

- Changing **views** (inside the same category page) must **not** reload
  the page---only blocks refetch.

- Clicking a **subcategory** navigates to a **new page** (expected).

- SSR is only for the initial render; subsequent view changes are
  client-side.

**Where enforced**

- handleChangeView in \[\...slug\].tsx uses fetch + router.push(\..., {
  shallow: true }).

## **Step 8 --- Testing checklist (frontend)**

**Explanation\**
QA can validate behavior quickly.

- **Initial SSR\**

  - GET /c/software-development returns HTML with meta
    title/description, breadcrumbs, subcategories, blocks.

- **View switch (no reload)\**

  - Click "Projects" → URL changes to /c/software-development/projects
    (shallow), blocks swap, header/breadcrumbs/subcategories stay.

  - Back to "All" → only blocks swap.

- **Hierarchy navigation\**

  - Click an L2 subcategory tile → full navigation to
    /c/software-development/blockchain-development.

  - Click an L3 subcategory tile →
    /c/software-development/blockchain-development/smart-contracts.

- **Skeletons\**

  - While switching views, the blocks area shows skeletons; the rest of
    the page remains.

- **Carousel policy\**

  - Sections marked carousel show arrows and page indicator; cannot
    exceed payload (API already capped).

- **RTL\**

  - Toggle useRtl(true) (or language switch) → layout flips to
    dir=\"rtl\" without breaking controls.
