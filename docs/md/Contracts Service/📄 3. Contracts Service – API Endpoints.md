## **4. Controllers & Services**

### **4.1 Contracts Controller**

File: /src/controllers/contracts.controller.ts

import { Controller, Post, Put, Get, Param, Body } from
\'@nestjs/common\';

import { ContractsService } from \'../services/contracts.service\';

import { EndContractDto } from \'../dtos/end-contract.dto\';

import { AcceptRejectDto } from \'../dtos/accept-reject.dto\';

\@Controller(\'contracts\')

export class ContractsController {

constructor(private readonly contractsService: ContractsService) {}

// End a single contract

\@Put(\':id/end\')

async endContract(@Param(\'id\') id: string, \@Body() dto:
EndContractDto) {

return this.contractsService.endContract(id, dto);

}

// Accept/Reject flow for talent

\@Put(\':id/respond\')

async respond(@Param(\'id\') id: string, \@Body() dto: AcceptRejectDto)
{

return this.contractsService.respondToOffer(id, dto);

}

// Get one contract by ID

\@Get(\':id\')

async getContract(@Param(\'id\') id: string) {

return this.contractsService.getContract(id);

}

// List contracts for a client

\@Get(\'client/:clientId\')

async getClientContracts(@Param(\'clientId\') clientId: string) {

return this.contractsService.getContractsByClient(clientId);

}

// List contracts for a talent

\@Get(\'talent/:talentId\')

async getTalentContracts(@Param(\'talentId\') talentId: string) {

return this.contractsService.getContractsByTalent(talentId);

}

}

## **5. Services (Core Logic)**

### **5.1 Contracts Service**

File: /src/services/contracts.service.ts

import { Injectable, NotFoundException } from \'@nestjs/common\';

import { EndContractDto } from \'../dtos/end-contract.dto\';

import { AcceptRejectDto } from \'../dtos/accept-reject.dto\';

import { ContractStatus } from \'../entities/enums\';

\@Injectable()

export class ContractsService {

async endContract(id: string, dto: EndContractDto) {

// 1. Fetch contract

// 2. Ensure status is ACTIVE

// 3. Call Billing Service for final reconciliation (emit event)

// 4. Mark status as ENDED and save

// 5. Emit \`contract.ended\` event (for Reviews + Notifications)

return { id, status: ContractStatus.ENDED, reason: dto.reasonCode };

}

async respondToOffer(id: string, dto: AcceptRejectDto) {

// Talent can accept or reject the pending contract

if (dto.accepted) {

// In reality, activation still depends on group deposit

return { id, message: \'Offer accepted. Waiting for deposit.\' };

} else {

return { id, status: ContractStatus.CANCELED, message: \'Offer
rejected.\' };

}

}

return { id, status: \'active\' }; // placeholder

export class WebhooksService {

// 1. Mark group.depositPaid = true

// 2. Activate all pending contracts in the group

// 3. Emit \`contract.activated\` events

return { groupId: payload.groupId, status: \'deposit cleared, contracts
activated\' };

}

// 1. Update contract billing records

// 2. Mark contract as ended if billing closed

return { contractId: payload.contractId, status: \'billing finalized\'
};

}

}

## **6. Summary of Endpoints**

  **Method**   **Path**                             **Purpose**
  ------------ ------------------------------------ -----------------------------------------
  POST         /contract-groups                     Create a group with multiple contracts
  GET          /contract-groups/:id                 Fetch group and all contracts
  PUT          /contracts/:id/end                   End one contract
  PUT          /contracts/:id/respond               Talent accepts/rejects contract
  GET          /contracts/:id                       Get one contract
  GET          /contracts/client/:id                Get all contracts for client
  GET          /contracts/talent/:id                Get all contracts for talent
  POST         /webhooks/checkout/deposit-cleared   Deposit cleared (Checkout → Contracts)
  POST         /webhooks/billing/finalized          Billing finalized (Billing → Contracts)
